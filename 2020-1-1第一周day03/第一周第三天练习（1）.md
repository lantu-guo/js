- 1、
```
    var i = 5;
    function fn(i) {
        return function (n) {
            console.log(n + (++i));  
        }
    }
    var f = fn(3);
    f(4);  //8
    fn(3)(6);  //10
    f(2);  //7
    console.log(i);  //5
```
i=5
f=fn(3)--->i=3 4,return
f(4)---->n=4,i=3--->4+4
fn(3)(6)--->i=3  4,n=6--->6+4
f(2)--->i=4 5,n=2--->2+5
